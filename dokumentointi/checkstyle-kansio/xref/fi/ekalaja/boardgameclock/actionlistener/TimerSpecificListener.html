7   import fi.ekalaja.boardgameclock.timers.SimpleTimer;
8   import java.awt.event.ActionEvent;
9   import java.awt.event.ActionListener;
10  import javax.swing.JButton;
11  
12  /**
13   * This class gives every created SimpleTimer two buttons,  
14   * which effect only that SimpleTimer.
15   */
16  public class TimerSpecificListener implements ActionListener {
17  
18      private SimpleTimer simpletimer;
19      private JButton addTime;
20      private JButton timePenalty;
21  
22  
23  /**
24   * This constructor is given the specific timer and two buttons related to it.
25   * @param simpletimer the specific timer
26   * @param addTime gives 15 seconds to the SimpleTimer
27   * @param timePenalty takes 15 seconds from the SimpleTimer
28   */
29      public TimerSpecificListener(SimpleTimer simpletimer, JButton addTime, JButton timePenalty) {
30          this.simpletimer = simpletimer;
31          this.addTime = addTime;
32          this.timePenalty = timePenalty;
33  
34      }
35  
36      @Override
37      public void actionPerformed(ActionEvent e) {
38          if (e.getSource() == addTime) {
39              simpletimer.addAQuarterMinute();
40              simpletimer.refreshFrameNumbers();
41          } else if (e.getSource() == timePenalty) {
42              simpletimer.reduceAQuarterMinute();
43              simpletimer.refreshFrameNumbers();
44          }
45  
46      }
47  
48  }
