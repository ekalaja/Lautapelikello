2    * This class is an ActionListener for the buttons and TextFields of the Setup

6   package fi.ekalaja.boardgameclock.actionlistener;
7   
8   import fi.ekalaja.boardgameclock.timers.ClockGroup;
9   import fi.ekalaja.boardgameclock.timelogic.TimersLogic;
10  import fi.ekalaja.boardgameclock.clockui.SwingUi;
11  import fi.ekalaja.boardgameclock.timelogic.HourglassLogic;
12  import java.awt.event.ActionEvent;
13  import java.awt.event.ActionListener;
14  import javax.swing.JButton;
15  import javax.swing.JTextField;
16  
17  public class SetupActionListener implements ActionListener {
18  
19      private JButton begin;
20      private JTextField numberOfClocks;
21      private JTextField givenMinutes;
22      private JTextField givenSeconds;
23      private JTextField givenExtraTime;
24      private ClockGroup clocks;
25      private final SwingUi swingui;
26      private final JButton hourglass;
27     
28  
29      /**
30       * This constructor inserts listener for buttons needed at Card1, which is
31       * setup card.
32       *
33       * @param begin starts a regular game
34       * @param hourglass starts 1v1 with hourglass game mode
35       * @param numberOfClocks number of clocks
36       * @param givenMinutes inserted minutes
37       * @param givenSeconds inserted seconds
38       * @param extraTime Additional time for every turn.
39       * @param swingui User interface being used
40       */
41      public SetupActionListener(JButton begin, JButton hourglass, JTextField numberOfClocks, JTextField givenMinutes, JTextField givenSeconds, JTextField extraTime, SwingUi swingui) {
42          this.swingui = swingui;
43          this.begin = begin;
44          this.hourglass = hourglass;
45          this.numberOfClocks = numberOfClocks;
46          this.givenMinutes = givenMinutes;
47          this.givenSeconds = givenSeconds;
48          this.givenExtraTime = extraTime;
49      }
50  
51      @Override
52      public void actionPerformed(ActionEvent ae) {
53          if (ae.getSource() == begin) {
54              int value = 0;
55              int minutes = 0;
56              int seconds = 0;
57              int extraTime = 0;
58  
59              try {
60                  
61                  value = Integer.parseInt(numberOfClocks.getText());
62                  minutes = Integer.parseInt(givenMinutes.getText());
63                  seconds = Integer.parseInt(givenSeconds.getText());
64                  extraTime = Integer.parseInt(givenExtraTime.getText());
65                  if (ae.getSource() == begin) {
66                      if ((value > 0 && minutes >= 0 && seconds >= 0 && extraTime >= 0)) {
67                          this.createClocksAndTimeLogic(value, minutes, seconds, extraTime);
68                      }
69                  }
70              } catch (Exception e) {
71                  System.out.println("Give positive integer values for time and players. Decimal figures are not allowed.");
72              }
73          } else if (ae.getSource() == hourglass) {
74              int minutes = 0;
75              int seconds = 0;
76              try {
77                  seconds = Integer.parseInt(givenSeconds.getText());
78                  minutes = Integer.parseInt(givenMinutes.getText());
79              } catch (Exception e) {
80                  System.out.println("Value of minutes must be positive. Decimal figures are not allowed.");
81              }
82              if ((minutes > 0 && seconds >= 0) | (minutes >= 0 && seconds > 0)) {
83                  this.createHourglassLogic(minutes, seconds);
84              }
85          }
86  
87      }
88  
89      /**
90       * This method is chosen if game mode is normal. It creates as many clocks
91       * as user has chosen in Card1.
92       *
93       * @param arvo number of players
94       * @param minutes amount of minutes in the beginning
95       * @param seconds amount of seconds in the beginning
96       * @param extraTime additional time for each turn
97       */
98      public void createClocksAndTimeLogic(int arvo, int minutes, int seconds, int extraTime) {
99          clocks = new ClockGroup();
100 
101         for (int i = 0; i < arvo; i++) {
102             clocks.addAClock(minutes, seconds);
103         }
104 
105         TimersLogic logic = new TimersLogic(clocks, extraTime);
106         swingui.addTimeLogicAndClocks(logic, clocks);
107         swingui.createCardTwo();
108 
109     }
110 
111     /**
112      * This method is chosen if game mode is hourglass. It creates
113      * HourglassLogic which always takes time from another player and gives it's
114      * opponent.
115      *
116      * @param minutes Value of full minutes used in "hourglass".
117      * @param seconds Value of seconds used in "hourglass".
118      */
119     public void createHourglassLogic(int minutes, int seconds) {
120         clocks = new ClockGroup();
121         clocks.addAClock(minutes, seconds);
122         clocks.addAClock(minutes, seconds);
123 
124         HourglassLogic hourglassL = new HourglassLogic(clocks);
125         swingui.addHourglassLogic(hourglassL, clocks);
126         swingui.createCardTwo();
127     }
128 
129 }
